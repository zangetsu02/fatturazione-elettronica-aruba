# XML Builder

Build, validate and serialize FatturaPA electronic invoices.

## FatturaBuilder

Fluent builder for invoice construction.

```typescript
import { FatturaBuilder } from '@fatturazione-elettronica-aruba/xml-builder';

const builder = FatturaBuilder.create();
```

### Methods

#### create()

Creates a new builder instance.

```typescript
const builder = FatturaBuilder.create();
```

#### setVersione(versione)

Sets the schema version (default: `'1.2.2'`).

```typescript
builder.setVersione('1.2.2');
```

#### setDatiTrasmissione(dati)

Sets complete transmission data.

```typescript
builder.setDatiTrasmissione({
  idTrasmittente: { idPaese: 'IT', idCodice: '01234567890' },
  progressivoInvio: '00001',
  formatoTrasmissione: 'FPR12',
  codiceDestinatario: 'ABC1234',
  pecDestinatario?: string,
  contattiTrasmittente?: { telefono?: string, email?: string },
});
```

#### setTrasmissioneB2B(params)

Helper for B2B transmission (FPR12 format).

```typescript
builder.setTrasmissioneB2B({
  idPaese: string;
  idCodice: string;
  progressivoInvio: string;
  codiceDestinatario?: string;  // Default: '0000000'
  pecDestinatario?: string;
});
```

#### setTrasmissionePA(params)

Helper for PA transmission (FPA12 format).

```typescript
builder.setTrasmissionePA({
  idPaese: string;
  idCodice: string;
  progressivoInvio: string;
  codiceDestinatario: string;   // 6-char IPA code
});
```

#### setCedentePrestatore(cedente)

Sets the seller/provider.

```typescript
builder.setCedentePrestatore({
  datiAnagrafici: {
    idFiscaleIVA?: { idPaese: string, idCodice: string },
    codiceFiscale?: string,
    anagrafica: { denominazione?: string, nome?: string, cognome?: string },
    regimeFiscale: RegimeFiscale,
  },
  sede: Indirizzo,
  stabileOrganizzazione?: Indirizzo,
  iscrizioneREA?: IscrizioneREA,
  contatti?: Contatti,
  riferimentoAmministrazione?: string,
});
```

#### setCessionarioCommittente(cessionario)

Sets the buyer/customer.

```typescript
builder.setCessionarioCommittente({
  datiAnagrafici: {
    idFiscaleIVA?: { idPaese: string, idCodice: string },
    codiceFiscale?: string,
    anagrafica: { denominazione?: string, nome?: string, cognome?: string },
  },
  sede: Indirizzo,
  stabileOrganizzazione?: Indirizzo,
  rappresentanteFiscale?: RappresentanteFiscale,
});
```

#### setDatiGenerali(dati)

Sets general document data.

```typescript
builder.setDatiGenerali({
  datiGeneraliDocumento: {
    tipoDocumento: TipoDocumento,
    divisa: string,           // E.g.: 'EUR'
    data: string,             // YYYY-MM-DD
    numero: string,
    importoTotaleDocumento?: number,
    causale?: string[],
    datiRitenuta?: DatiRitenuta,
    datiBollo?: DatiBollo,
  },
  datiOrdineAcquisto?: DatiDocumentoCorrelato[],
  datiContratto?: DatiDocumentoCorrelato[],
  datiDDT?: DatiDDT[],
});
```

#### setDatiBeniServizi(dati)

Sets line items and VAT summary.

```typescript
builder.setDatiBeniServizi({
  dettaglioLinee: [
    {
      numeroLinea: number,
      descrizione: string,
      quantita?: number,
      unitaMisura?: string,
      prezzoUnitario: number,
      prezzoTotale: number,
      aliquotaIVA: number,
      natura?: Natura,
      scontoMaggiorazione?: ScontoMaggiorazione[],
    },
  ],
  datiRiepilogo: [
    {
      aliquotaIVA: number,
      imponibileImporto: number,
      imposta: number,
      natura?: Natura,
      esigibilitaIVA?: 'I' | 'D' | 'S',
      riferimentoNormativo?: string,
    },
  ],
});
```

#### addLinea(linea)

Adds a single line item. `numeroLinea` (incremental) and `prezzoTotale`
(`prezzoUnitario` × `quantita`, applying discounts) are computed if omitted.
`DatiRiepilogo` and `ImportoTotaleDocumento` are computed automatically in `build()`.

```typescript
builder.addLinea({
  descrizione: string,
  quantita?: number,
  unitaMisura?: string,
  prezzoUnitario: number,
  aliquotaIVA: number,
  natura?: Natura,
  scontoMaggiorazione?: ScontoMaggiorazione[],
  numeroLinea?: number,   // optional, auto-assigned
  prezzoTotale?: number,  // optional, auto-computed
});
```

#### setDettaglioLinee(linee)

Sets all line items at once (replaces previously added lines). Same auto-computation as `addLinea`.

```typescript
builder.setDettaglioLinee([
  { descrizione: 'Item A', quantita: 2, prezzoUnitario: 50, aliquotaIVA: 22 },
  { descrizione: 'Item B', quantita: 1, prezzoUnitario: 100, aliquotaIVA: 10 },
]);
```

#### setEsigibilitaIVA(esigibilitaIVA)

Sets the default `EsigibilitaIVA` (`'I'` | `'D'` | `'S'`, default `'I'`) applied to
the auto-computed `DatiRiepilogo`. Not applied to groups with a `Natura`.

```typescript
builder.setEsigibilitaIVA('S'); // split payment
```

#### addDatiPagamento(pagamento)

Adds payment data.

```typescript
builder.addDatiPagamento({
  condizioniPagamento: 'TP01' | 'TP02' | 'TP03',
  dettaglioPagamento: [
    {
      modalitaPagamento: ModalitaPagamento,
      importoPagamento: number,
      dataScadenzaPagamento?: string,
      iban?: string,
      istitutoFinanziario?: string,
    },
  ],
});
```

#### newBody()

Finalizes current body and starts a new one (for batch invoices).

```typescript
builder
  .setDatiGenerali({ ... })
  .setDatiBeniServizi({ ... })
  .newBody()
  .setDatiGenerali({ ... })  // New invoice
  .setDatiBeniServizi({ ... });
```

#### build()

Builds the electronic invoice. Repeatable: it does not mutate the builder state.

```typescript
const fattura: FatturaElettronica = builder.build();
```

#### validate(options?)

Builds and validates in one call, returning a `ValidationResult`.

```typescript
const result = builder.validate({ strict: true });
```

#### toXml(options?)

Builds and serializes to FatturaPA XML in one call.

```typescript
const xml = builder.toXml({ includeSchemaLocation: true });
```

#### toResult(options?)

Builds, validates and serializes in one call, also returning the SDI filename.

```typescript
const { fattura, xml, validazione, filename } = builder.toResult({
  validation?: ValidationOptions,
  serializer?: FatturaSerializerOptions,
});
// filename: '<IdPaese><IdCodice>_<Progressivo>.xml'
```

#### reset()

Resets the builder to initial state.

```typescript
builder.reset();
```

## FatturaValidator

Validator for electronic invoices.

```typescript
import { FatturaValidator, validateFattura } from '@fatturazione-elettronica-aruba/xml-builder';

// Instance
const validator = new FatturaValidator(options);
const result = validator.validate(fattura);

// Helper
const result = validateFattura(fattura, options);
```

### Options

```typescript
interface ValidationOptions {
  validateTotals?: boolean;        // Default: true
  validateCodiceFiscale?: boolean; // Default: true
  validatePartitaIVA?: boolean;    // Default: true
  validateDates?: boolean;         // Default: true
  strict?: boolean;                // Default: false
}
```

### Result

```typescript
interface ValidationResult {
  valid: boolean;
  errors: ValidationError[];
  warnings: ValidationError[];
}

interface ValidationError {
  path: string;
  message: string;
  code: string;
}
```

## FatturaSerializer

Serializes invoices to XML.

```typescript
import { FatturaSerializer } from '@fatturazione-elettronica-aruba/xml-builder';

const serializer = new FatturaSerializer({
  includeSchemaLocation?: boolean,  // Default: true
});

const xml = serializer.serialize(fattura);
```

::callout
The serializer coerces values defensively: numbers passed where strings are
expected (e.g. VAT numbers from a database) are serialized without throwing.
`Quantita` keeps 2–8 decimals (e.g. `0.001`), prices/VAT rates use 2 decimals.
::

## Utilities

Pure helpers exported from the package. Useful to map domain data without
re-implementing VAT/total logic or nested anagrafica objects.

### calcolaRiepilogo(linee, options?)

Computes `DatiRiepilogo` (grouped by `aliquotaIVA` + `natura`) and the
`importoTotaleDocumento` from the line items, with correct rounding (SDI rules).

```typescript
const { datiRiepilogo, importoTotaleDocumento } = calcolaRiepilogo(linee, {
  esigibilitaIVA: 'I', // default; not applied to groups with a Natura
});
```

### calcolaPrezzoTotale(prezzoUnitario, quantita?, scontoMaggiorazione?)

Computes a line's `PrezzoTotale` (= `prezzoUnitario` × `quantita`, applying
discounts/surcharges), rounded to 2 decimals.

```typescript
calcolaPrezzoTotale(75, 40);                              // 3000
calcolaPrezzoTotale(100, 1, [{ tipo: 'SC', percentuale: 10 }]); // 90
```

### toNumber(value) / round2(value)

`toNumber` coerces `number | string | null | undefined` to a number (DB numerics
often arrive as strings). `round2` rounds to 2 decimals avoiding floating-point errors.

```typescript
toNumber('12.50'); // 12.5
round2(1.005);     // 1.01
```

### cedentePrestatore(input) / cessionario(input)

Build `CedentePrestatore` / `CessionarioCommittente` from flat fields, with string
coercion and optional `IdFiscaleIVA`/`CodiceFiscale`/contacts.

```typescript
const cedente = cedentePrestatore({
  piva: '01234567890', codiceFiscale, denominazione: 'Azienda SRL',
  regimeFiscale: 'RF01',           // default 'RF01'
  indirizzo, cap, comune, provincia, nazione, // nazione default 'IT'
  telefono, email,
});

const cliente = cessionario({
  piva, codiceFiscale,            // both optional
  denominazione, indirizzo, cap, comune, provincia,
});
```

### sede(input) / contatti(input)

`sede` builds an `Indirizzo` (default `nazione: 'IT'`); `contatti` builds a
`Contatti` block or returns `undefined` if all fields are empty.

### nomeFileSdi(idPaese, idCodice, progressivo)

Builds the SDI filename `<IdPaese><IdCodice>_<Progressivo>.xml` (progressivo
left-padded to 5 chars).

```typescript
nomeFileSdi('IT', '01234567890', '1'); // 'IT01234567890_00001.xml'
```
